```python
# ======================================================
# NUEVO: Descargar plantilla oficial
# ======================================================

def download_template(self):
    """
    Permite al usuario descargar una copia de la plantilla oficial.

    La plantilla original queda protegida dentro del sistema.
    El usuario solo recibe una copia para llenar.
    """

    try:
        source_path = get_template_excel_path()

        suggested_name = "plantilla_homologacion_items_issfa.xlsx"

        destination_path, _ = QFileDialog.getSaveFileName(
            self,
            "Guardar plantilla Excel",
            suggested_name,
            "Excel (*.xlsx)",
        )

        if not destination_path:
            return

        if not destination_path.lower().endswith(".xlsx"):
            destination_path += ".xlsx"

        shutil.copyfile(source_path, destination_path)

        QMessageBox.information(
            self,
            "Plantilla descargada",
            "La plantilla fue descargada correctamente.\n\n"
            "Ahora el usuario debe llenarla y luego cargarla con el botón "
            "'Seleccionar Excel'."
        )

    except Exception as exc:
        QMessageBox.critical(
            self,
            "Error descargando plantilla",
            str(exc)
        )

# ======================================================
```

```python
# scripts/create_template.py
# ======================================================
# Crea la plantilla oficial de homologación ISSFA
# ======================================================

from pathlib import Path
from openpyxl import Workbook
from openpyxl.styles import Font, PatternFill, Alignment
from openpyxl.worksheet.datavalidation import DataValidation


ROOT = Path(__file__).resolve().parent.parent
OUTPUT = ROOT / "resources" / "templates" / "plantilla_homologacion_items_issfa.xlsx"


def main():
    OUTPUT.parent.mkdir(parents=True, exist_ok=True)

    wb = Workbook()

    ws = wb.active
    ws.title = "Plantilla"

    headers = [
        "CODIGO_ACTUAL",
        "DESCRIPCION_ACTUAL",
        "CODIGO_NUEVO",
        "DESCRIPCION_NUEVA",
    ]

    ws.append(headers)

    header_fill = PatternFill("solid", fgColor="D9EAF7")
    header_font = Font(bold=True)
    center = Alignment(horizontal="center", vertical="center", wrap_text=True)

    for col in range(1, len(headers) + 1):
        cell = ws.cell(row=1, column=col)
        cell.fill = header_fill
        cell.font = header_font
        cell.alignment = center

    widths = {
        "A": 22,
        "B": 60,
        "C": 22,
        "D": 60,
    }

    for col, width in widths.items():
        ws.column_dimensions[col].width = width

    ws.freeze_panes = "A2"

    # Filas vacías para que el usuario llene.
    for row in range(2, 502):
        for col in range(1, 5):
            ws.cell(row=row, column=col).alignment = Alignment(wrap_text=True, vertical="top")

    # Hoja de ejemplo.
    ex = wb.create_sheet("Ejemplo")
    ex.append(headers)
    ex.append([
        "010101",
        "CONSULTA MEDICA GENERAL",
        "010101-A",
        "CONSULTA MÉDICA GENERAL",
    ])
    ex.append([
        "020202",
        "RADIOGRAFIA TORAX",
        "020202-A",
        "RADIOGRAFÍA DE TÓRAX",
    ])

    for col in range(1, len(headers) + 1):
        cell = ex.cell(row=1, column=col)
        cell.fill = header_fill
        cell.font = header_font
        cell.alignment = center

    for col, width in widths.items():
        ex.column_dimensions[col].width = width

    ex.freeze_panes = "A2"

    # Hoja de instrucciones.
    ins = wb.create_sheet("Instrucciones")
    instructions = [
        ["Campo", "Descripción"],
        ["CODIGO_ACTUAL", "Código que actualmente existe en Oracle."],
        ["DESCRIPCION_ACTUAL", "Descripción actual esperada del código existente."],
        ["CODIGO_NUEVO", "Nuevo código que reemplazará al código actual o será insertado."],
        ["DESCRIPCION_NUEVA", "Nueva descripción que se desea registrar."],
        ["Importante", "No cambiar los nombres de las columnas."],
        ["Importante", "No eliminar la fila 1."],
        ["Importante", "El sistema mostrará una vista previa antes de aplicar cambios."],
    ]

    for row in instructions:
        ins.append(row)

    ins.column_dimensions["A"].width = 28
    ins.column_dimensions["B"].width = 90

    for col in range(1, 3):
        cell = ins.cell(row=1, column=col)
        cell.fill = header_fill
        cell.font = header_font
        cell.alignment = center

    wb.save(OUTPUT)

    print(f"Plantilla creada en: {OUTPUT}")


if __name__ == "__main__":
    main()
```

## Cambio 1: agregar función para construir resumen

En `app/main.py`, agregar esta función fuera de la clase `MainWindow`:

```python
# ======================================================
# NUEVO: Resumen previo de acciones en base de datos
# ======================================================

def build_db_action_summary(rows: list[dict]) -> str:
    """
    Construye un resumen claro antes de aplicar cambios.
    Este texto se muestra al usuario antes del COMMIT.

    No ejecuta SQL.
    Solo resume lo que se hará según la vista previa.
    """

    selected = [r for r in rows if r.get("APLICAR")]

    updates = [r for r in selected if r.get("ACCION") == "UPDATE"]
    inserts = [r for r in selected if r.get("ACCION") == "INSERT"]
    blocked = [r for r in rows if r.get("ACCION") == "BLOQUEADO"]
    not_selected = [r for r in rows if not r.get("APLICAR")]

    lines = []

    lines.append("RESUMEN DE ACCIONES EN BASE DE DATOS")
    lines.append("")
    lines.append("Tabla afectada:")
    lines.append("  - SIS.ITEMS_ISSFA_DETALLE")
    lines.append("")
    lines.append("Acciones seleccionadas:")
    lines.append(f"  - UPDATE: {len(updates)} registro(s)")
    lines.append(f"  - INSERT: {len(inserts)} registro(s)")
    lines.append(f"  - NO SE APLICAN / DESMARCADOS: {len(not_selected)} registro(s)")
    lines.append(f"  - BLOQUEADOS: {len(blocked)} registro(s)")
    lines.append("")
    lines.append("Protecciones:")
    lines.append("  - No se tocará SIS.ITEMS_ISSFA_CABECERA.")
    lines.append("  - No se aplicarán filas bloqueadas.")
    lines.append("  - No se aplicarán filas desmarcadas.")
    lines.append("  - Se usará una sola transacción.")
    lines.append("  - Si ocurre un error, se ejecutará ROLLBACK completo.")
    lines.append("")

    if updates:
        lines.append("DETALLE DE UPDATE:")
        for r in updates[:30]:
            lines.append(
                "  - "
                f"{r.get('CODIGO_ACTUAL')} → {r.get('CODIGO_NUEVO')} | "
                f"{r.get('DESCRIPCION_ACTUAL_ORACLE') or r.get('DESCRIPCION_ACTUAL_EXCEL')} "
                f"→ {r.get('DESCRIPCION_NUEVA')}"
            )

        if len(updates) > 30:
            lines.append(f"  ... y {len(updates) - 30} UPDATE adicional(es).")

        lines.append("")

    if inserts:
        lines.append("DETALLE DE INSERT:")
        for r in inserts[:30]:
            lines.append(
                "  - "
                f"{r.get('CODIGO_NUEVO')} | "
                f"{r.get('DESCRIPCION_NUEVA')} | "
                f"TIPO: {r.get('TIPO')}"
            )

        if len(inserts) > 30:
            lines.append(f"  ... y {len(inserts) - 30} INSERT adicional(es).")

        lines.append("")

    if blocked:
        lines.append("FILAS BLOQUEADAS:")
        for r in blocked[:30]:
            lines.append(
                "  - "
                f"Fila Excel {r.get('FILA_EXCEL')} | "
                f"Código actual: {r.get('CODIGO_ACTUAL')} | "
                f"Código nuevo: {r.get('CODIGO_NUEVO')} | "
                f"Motivo: {r.get('STATUS')}"
            )

        if len(blocked) > 30:
            lines.append(f"  ... y {len(blocked) - 30} bloqueada(s) adicional(es).")

        lines.append("")

    lines.append("Confirmación requerida:")
    lines.append("  El usuario debe aceptar explícitamente para aplicar los cambios.")

    return "\n".join(lines)

# ======================================================
# FIN NUEVO
# ======================================================
```

```python
def start_apply(self):
    """
    Antes de aplicar cambios, sincroniza las filas marcadas,
    construye un resumen técnico y solicita confirmación explícita.
    """

    self.sync_checked_rows()

    selected = [r for r in self.rows if r.get("APLICAR")]

    if not selected:
        QMessageBox.warning(
            self,
            "Validación",
            "No hay filas seleccionadas para aplicar.",
        )
        return

    summary = build_db_action_summary(self.rows)

    confirm = QMessageBox.question(
        self,
        "Confirmar acciones en base de datos",
        summary + "\n\n¿Desea continuar con la aplicación de cambios?",
        QMessageBox.StandardButton.Yes | QMessageBox.StandardButton.No,
        QMessageBox.StandardButton.No,
    )

    if confirm != QMessageBox.StandardButton.Yes:
        return

    second_confirm = QMessageBox.question(
        self,
        "Confirmación final",
        "Esta operación modificará datos en Oracle.\n\n"
        "Se ejecutará dentro de una transacción.\n"
        "Si ocurre un error, se hará rollback completo.\n\n"
        "¿Confirma definitivamente la aplicación?",
        QMessageBox.StandardButton.Yes | QMessageBox.StandardButton.No,
        QMessageBox.StandardButton.No,
    )

    if second_confirm != QMessageBox.StandardButton.Yes:
        return

    audit_dir = Path("auditoria")
    audit_name = datetime.now().strftime("homologacion_%Y%m%d_%H%M%S.csv")
    audit_path = str(audit_dir / audit_name)

    self.apply_button.setEnabled(False)
    self.analyze_button.setEnabled(False)
    self.info_label.setText("Aplicando cambios en Oracle...")

    self.apply_worker = ApplyWorker(
        conn=self.conn,
        rows=self.rows,
        audit_path=audit_path,
    )
    self.apply_worker.finished_ok.connect(self.apply_ok)
    self.apply_worker.failed.connect(self.apply_failed)
    self.apply_worker.start()
```

# 8. Instrucción exacta para el programador

Antes de ejecutar cualquier `UPDATE` o `INSERT`, el sistema debe construir y mostrar un resumen técnico de acciones sobre Oracle. Ese resumen debe indicar cuántos registros serán actualizados, insertados, omitidos o bloqueados, y debe listar los códigos afectados.

El usuario debe aceptar explícitamente ese resumen. Luego debe existir una segunda confirmación final indicando que se modificará Oracle dentro de una transacción con rollback completo si ocurre un error.

No se debe ejecutar ningún cambio en base de datos si el usuario cierra, cancela o rechaza cualquiera de las dos confirmaciones.
